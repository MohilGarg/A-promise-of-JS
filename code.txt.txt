async function respond(inputText){
	var weight = 0;
	var input = inputText.toLowerCase();
	for(var i=0; i<input.length; ++i)
	{
		weight = weight + Number(input.charCodeAt(i)-96);
	}
	Bot.send("Weight of the word is : " + weight);
	if(weight % 2 == 0)
		Bot.send("Even Number");
	else
		Bot.send("Odd Number");
 }
